import pandas as pd
import re

header = {}
with open("01.dat", "r") as f:
    for line in f:
      if r := re.match(r'^INFO,([^,]*),([^,]*)$', line):
        header[r[2].strip()] = r[1]
      if line.strip() == "[Data]":
        break
    df = pd.read_csv(f, delimiter=",", index_col="Temperature (K)")
df['Xi'] = df['Moment (emu)'] / df['Magnetic Field (Oe)']

```{python}
import pandas as pd
import re

header = {}
with open("01.dat", "r") as f:
    for line in f:
      if r := re.match(r'^INFO,([^,]*),([^,]*)$', line):
        header[r[2].strip()] = r[1]
      if line.strip() == "[Data]":
        break
    df = pd.read_csv(f, delimiter=",", index_col="Temperature (K)")
df['Xi'] = df['Moment (emu)'] / df['Magnetic Field (Oe)'] 
pd.options.plotting.backend = "plotly"
df.plot(y='Xi')
```
